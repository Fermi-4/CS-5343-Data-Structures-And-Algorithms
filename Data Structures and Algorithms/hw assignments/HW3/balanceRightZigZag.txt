balanceRightZigZag(n1, n2, n3) {
	
	// rotate 2, 3
	n2.left = n3.right;
	n3.right.parent = n2;
	n3.right = n2;
	n2.parent = n3;
	n3.parent = n1;
	n1.right = n3;
	
	// balance right straight line
	n1.parent.right = n3;
	n1.parent = n3;
	tmp = n3.left;
	tmp.parent = n1;
	n3.left = n1;
	n1.right = tmp;
}
